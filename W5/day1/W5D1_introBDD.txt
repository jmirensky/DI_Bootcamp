ELIMINAR un registro

id NO SE reinicia

ELIMINAR un registro con clausula de retorno

WHERE condition
RETURNING variable1, variable2...;



TRUNCATE TABLE table_name y DELETE FROM table_name se usan para lo mismo, pero hay diferencias:

Para eliminar todas las filas de una tabla,  omitir la cláusula WHERE
DELETE FROM table;
--no vuelve a id_pk = 1 (no restablece identidad)
-- es un comando de lenguaje de manipulación de datos (DML).

TRUNCAR una tabla = deja limpias todas las filas de la tabla. 
se puede borrar y al volver a insertar valores, elegir si queremos empezar desde 1 o continuar con la numeracion serial
TRUNCATE TABLE table_name RESTART IDENTITY; --> recupera el almacenamiento de inmediato.  reinicia la columna SERIAL.
-- es un comando de lenguaje de definición de datos (DDL).




ALTER TABLE --> ALTER TABLE table_name action;
Para cambiar la estructura de la tabla existente

Añadir una nueva columna
ALTER TABLE table_name ADD COLUMN new_column_name TYPE;

Eliminar una columna existente
ALTER TABLE table_name DROP COLUMN column_name;

Renombrar una columna existente
ALTER TABLE table_name RENAME COLUMN column_name TO new_column_name;

Renombrar una tabla
ALTER TABLE table_name RENAME TO new_table_name;

Cambia el tipo de dato de una columna
ALTER TABLE table_name
ALTER COLUMN column_name1 [SET DATA] TYPE new_data_type;
ALTER COLUMN column_name2 [SET DATA] TYPE new_data_type;

Para eliminar una tabla existente de la base de datos,
DROP TABLE IF EXISTS table_name;
